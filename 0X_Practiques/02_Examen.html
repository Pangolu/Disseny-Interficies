// Implementación de arrastrar y soltar
element.addEventListener('mousedown', iniciarArrastre);

function iniciarArrastre(e) {
    // Calcular desplazamiento del mouse
    offsetX = e.clientX - element.getBoundingClientRect().left;
    offsetY = e.clientY - element.getBoundingClientRect().top;
    
    // Mover elemento durante el arrastre
    document.addEventListener('mousemove', moverElemento);
    document.addEventListener('mouseup', soltarElemento);
}

function moverElemento(e) {
    // Calcular nueva posición
    const x = e.clientX - offsetX - container.getBoundingClientRect().left;
    const y = e.clientY - offsetY - container.getBoundingClientRect().top;
    
    // Aplicar nueva posición
    element.style.left = x + 'px';
    element.style.top = y + 'px';
}
